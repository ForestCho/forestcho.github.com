用dom4j读取和更新xml文件

2013-10-20
畅游互联网
dom4j, 操作xml, 配置文件

平时项目中，涉及到xml解析的操作很多，解析http请求，读取配置文件。解析XML的方法很多，有dom，sax等，作为一个初学者，我发现还是DOM4J比较适合我的使用习惯。他是一种开源的xml API，提供大量的API方法（不用我们自己去写），简单高效。
configure.java

public class configure {
    public String username=null;
    public String password=null;
    public void setUsername(String username){
        this.username=username;
    }
    public void setPassword(String password){
        this.password=password;
    }
    public String getUsername(){
        return this.username;
    }
    public String getPassword(){
        return this.password;
    }
}

readFile.java 操作xml数据的类，有读取xml和更新后保存xml的方法。 

import java.io.File;
import java.io.FileWriter;
import java.io.IOException;
import java.io.PrintWriter;
import java.io.Writer;
import java.util.List;
import java.util.Vector; 
import org.dom4j.Document;
import org.dom4j.DocumentException;
import org.dom4j.DocumentHelper;
import org.dom4j.Element;
import org.dom4j.io.OutputFormat; 
import org.dom4j.io.SAXReader;
import org.dom4j.io.XMLWriter;

public class readFile {
    public String filePath = "WebRoot\\WEB-INF\\test.xml";
    public String filePath1 = "WebRoot\\WEB-INF\\test1.xml";
    Vector confVec = null;

    public void writeXml() {
        try {
            Document doc =  DocumentHelper.createDocument(); 
            Element root = doc.addElement("configures"); 
            for (int i = 0; i < confVec.size(); i++) {
                configure conf = (configure) confVec.get(i);
                Element confE = root.addElement("configure").addAttribute("id", Integer.toString(i));

                Element usernameE = confE.addElement("username");
                usernameE.setText(conf.getUsername());

                Element passwordE = confE.addElement("password");
                passwordE.setText(conf.getPassword()); 
            }
               OutputFormat format = new OutputFormat("    ", true);
               Writer out=new FileWriter(filePath1);
               format.setEncoding("gb2312");
               // 可以把System.out改为你要的流。
               XMLWriter xmlWriter = new XMLWriter(out, format);
               xmlWriter.write(doc);
               xmlWriter.close();

        } catch (IOException e) {
            // TODO Auto-generated catch block
            e.printStackTrace();
        }
    }

    public void readXml() {
        try {
            confVec=new Vector();
            Document doc; 
            SAXReader reader=new SAXReader();
            doc = reader.read(new File(filePath));             
            Element root=doc.getRootElement(); 
            List confList=root.elements("configure");
            for (int i = 0; i < confList.size(); i++) {
                Element confE=(Element) confList.get(i);
                configure conf = new configure(); 
                if(i==1){
                    confE.element("username").setText("ccc");//此处进行了update
                }
                String username = confE.element("username").getText();            
                String password = confE.element("password").getText();
                 System.out.println(username+":"+password);
                 conf.setUsername(username);
                 conf.setPassword(password);
                confVec.add(conf);
            }
        }  catch (DocumentException e) {
            // TODO Auto-generated catch block
            e.printStackTrace();
        }
    }
}
